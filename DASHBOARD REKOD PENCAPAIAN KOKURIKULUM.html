      if (!dateStr) return '-';
      const date = new Date(dateStr);
      return date.toLocaleDateString('ms-MY', { day: 'numeric', month: 'long', year: 'numeric' });
    }

    // Render records

      const emptyState = document.getElementById('emptyState');
      const countSpan = document.querySelector('#recordCount span');

      if (records.length === 0) {
        emptyState.classList.remove('hidden');
        container.classList.add('hidden');
        return;
      }

      container.classList.remove('hidden');
      
      container.innerHTML = '';
      
      records.forEach((record, index) => {
        const teachers = [];

          const teacher = record[`guru${i}`];
          if (teacher && teacher.trim() !== '') {
            teachers.push(teacher);
          }

        const students = [];

          const student = record[`murid${i}`];
          if (student && student.trim() !== '') {
            students.push(student);
          }
        }
        
        const card = document.createElement('div');
        card.className = 'bg-black/40 border gold-border rounded-xl p-4 hover:bg-black/60 transition-colors';
        card.innerHTML = `

                <p class="text-amber-100">${record.penganjur || '-'}</p>

              <div>
                <span class="text-amber-400/70 text-xs">Pencapaian</span>

              </div>

      });

    // Show delete modal
    function showDeleteModal(id) {
      recordToDelete = id;
      document.getElementById('deleteModal').classList.remove('hidden');
    }

    // Hide delete modal
    function hideDeleteModal() {
      recordToDelete = null;
      document.getElementById('deleteModal').classList.add('hidden');
    }

    // Delete record
    async function deleteRecord() {
      if (!recordToDelete) return;
      
      const record = records.find(r => r.__backendId === recordToDelete);
      if (!record) {
        hideDeleteModal();
        return;
      }
      
      const confirmBtn = document.getElementById('confirmDelete');
      confirmBtn.disabled = true;
      confirmBtn.textContent = 'Memadam...';
      
      const result = await window.dataSdk.delete(record);
      
      confirmBtn.disabled = false;
      confirmBtn.textContent = 'Padam';
      hideDeleteModal();
      
      if (!result.isOk) {
        showToast('Gagal memadam rekod. Sila cuba lagi.', 'error');
      }
    }

    // Toast notification
    function showToast(message, type = 'success') {
      const toast = document.createElement('div');
      toast.className = `fixed bottom-4 right-4 px-6 py-3 rounded-lg shadow-lg z-50 transition-all duration-300 ${type === 'success' ? 'bg-green-900 border border-green-500 text-green-200' : 'bg-red-900 border border-red-500 text-red-200'}`;
      toast.textContent = message;
      document.body.appendChild(toast);
      
      setTimeout(() => {
        toast.style.opacity = '0';
        setTimeout(() => toast.remove(), 300);
      }, 3000);
    }
