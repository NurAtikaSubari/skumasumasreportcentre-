    function muatTurunPDFKelas() {
      const { jsPDF } = window.jspdf;
      const doc = new jsPDF();
      
      let filtered = allRecords.filter(r = r.jenis_rekod === 'rekod_kelas');
      
      if (filtered.length === 0) {
        showNotification('Tiada rekod untuk dimuat turun', 'error');
        return;
      }
      
      // Group by kelas
      const groupedByKelas = {};
      filtered.forEach(record = {
        if (!groupedByKelas[record.kelas]) {
          groupedByKelas[record.kelas] = [];
        }
        groupedByKelas[record.kelas].push(record);
      });
      
      // Sort each kelas by date descending
      Object.keys(groupedByKelas).forEach(kelas = {
        groupedByKelas[kelas].sort((a, b) = new Date(b.tarikh) - new Date(a.tarikh));
      });
      
      // Header
      doc.setFontSize(18);
      doc.setFont(undefined, 'bold');
      doc.text('E-REKOD KAWALAN KELAS', 105, 20, { align: 'center' });
      
      doc.setFontSize(12);
      doc.setFont(undefined, 'normal');
      doc.text('Sekolah Kebangsaan Umas-Umas, Tawau Sabah', 105, 28, { align: 'center' });
      doc.text(`Dijana pada: ${new Date().toLocaleDateString('ms-MY')}`, 105, 35, { align: 'center' });
      
      let yPos = 45;
      
      Object.keys(groupedByKelas).forEach(kelas = {
        if (yPos  240) {
          doc.addPage();
          yPos = 20;
        }
        
        doc.setFontSize(14);
        doc.setFont(undefined, 'bold');
        doc.text(`KELAS: ${kelas}`, 15, yPos);
        yPos += 8;
        
        doc.setFontSize(11);
        doc.setFont(undefined, 'normal');
        
        groupedByKelas[kelas].forEach((record, idx) = {
          if (yPos  240) {
            doc.addPage();
            yPos = 20;
          }
          
          doc.setFont(undefined, 'bold');
          doc.text(`${idx + 1}. ${record.nama_guru}`, 20, yPos);
          yPos += 5;
          
          doc.setFont(undefined, 'normal');
          doc.setFontSize(10);
          doc.text(`Subjek: ${record.subjek}`, 25, yPos);
          yPos += 4;
          doc.text(`Tarikh: ${formatTarikh(record.tarikh)} | Masa: ${record.masa_masuk} - ${record.masa_keluar}`, 25, yPos);
          yPos += 4;
          doc.text(`Kehadiran: ${record.jumlah_kehadiran} murid`, 25, yPos);
          yPos += 4;
          
          doc.setFont(undefined, 'bold');
          doc.text('Ulasan:', 25, yPos);
          yPos += 4;
          
          doc.setFont(undefined, 'normal');
          const ulasanLines = doc.splitTextToSize(record.ulasan_guru, 160);
          ulasanLines.forEach(line = {
            if (yPos  240) {
              doc.addPage();
              yPos = 20;
            }
            doc.text(line, 30, yPos);
            yPos += 4;
          });
          
          yPos += 6;
          doc.setDrawColor(200, 200, 200);
          doc.line(15, yPos, 195, yPos);
          yPos += 6;
        });
      });
      
      // Footer
      const pageCount = doc.internal.getNumberOfPages();
      for (let i = 1; i = pageCount; i++) {
        doc.setPage(i);
        doc.setFontSize(9);
        doc.setFont(undefined, 'normal');
        doc.text(`Halaman ${i} daripada ${pageCount}`, 105, 290, { align: 'center' });
      }
      
      const fileName = `E-Rekod_Kawalan_Kelas_${new Date().getFullYear()}.pdf`;
      doc.save(fileName);
      showNotification('PDF berjaya dimuat turun!', 'success');
    }
