window.filterLaporanBy = function(type) {

  const buttons = document.querySelectorAll(".filter-btn");
  buttons.forEach(btn => btn.classList.remove("active"));
  document.querySelector(`[data-filter="${type}"]`).classList.add("active");

  const items = document.querySelectorAll(".laporan-item");

  items.forEach(item => {
    if (type === "semua") {
      item.style.display = "block";
    } else {
      item.style.display =
        item.getAttribute("data-type") === type
          ? "block"
          : "none";
    }
  });
};

<script>
function printLaporan(type, tarikh, nama, jumlah, kelas, ulasan) {

  const { jsPDF } = window.jspdf;
  const doc = new jsPDF();
  const today = new Date().toLocaleDateString("ms-MY");

  doc.setFontSize(14);
  doc.text("SEKOLAH KEBANGSAAN UMAS-UMAS", 105, 20, { align: "center" });

  if (type === "rmt-guru") {
    doc.text("LAPORAN RMT GURU", 105, 30, { align: "center" });
  } else {
    doc.text("LAPORAN RMT MURID", 105, 30, { align: "center" });
  }

  doc.setFontSize(11);
  doc.text("Tarikh: " + tarikh, 20, 50);
  doc.text("Nama: " + nama, 20, 60);
  doc.text("Jumlah: " + jumlah, 20, 70);
  doc.text("Kelas: " + kelas, 20, 80);

  doc.text("Ulasan:", 20, 95);
  doc.text(ulasan || "-", 20, 105, { maxWidth: 170 });

  doc.text("Dicetak pada: " + today, 20, 280);

  doc.save("Laporan_" + tarikh + ".pdf");
}
