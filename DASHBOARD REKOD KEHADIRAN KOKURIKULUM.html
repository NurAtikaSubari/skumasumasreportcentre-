    // Function to check and delete old reports (after Dec 31)
    function checkAndDeleteOldReports(allRecords) {
      const now = new Date();
      const currentYear = now.getFullYear();
      const endOfYear = new Date(currentYear, 0, 1); // Jan 1 of current year
      
      // Only delete if we're in a new year and have records from previous year
      const recordsToDelete = allRecords.filter(record => {
        const recordDate = new Date(record.created_at);
        return recordDate < endOfYear;
      });

      // Delete old records
      recordsToDelete.forEach(async (record) => {
        await window.dataSdk.delete(record);
      });

      if (recordsToDelete.length > 0) {
        showToast(`${recordsToDelete.length} laporan lama telah dipadam (sebelum 1 Januari ${currentYear})`);
      }
    }

    // Function to render reports table
    function renderReportsTable(data) {
      const tbody = document.getElementById('reports-table-body');
      const noReports = document.getElementById('no-reports');
      const reportCount = document.getElementById('report-count');
      
      reportCount.textContent = data.length;
      
      if (data.length === 0) {
        tbody.innerHTML = '';
        noReports.classList.remove('hidden');
        return;
      }
      
      noReports.classList.add('hidden');
      
      // Sort by perjumpaan (ascending), then by date (newest first)
      const sortedData = [...data].sort((a, b) => {
        if (a.perjumpaan !== b.perjumpaan) {
          return a.perjumpaan - b.perjumpaan;
        }
        return new Date(b.created_at) - new Date(a.created_at);
      });
      
      let currentPerjumpaan = null;
      let html = '';
      
      sortedData.forEach((record, index) => {
        const recordDate = new Date(record.tarikh);
        const formattedDate = recordDate.toLocaleDateString('ms-MY', { day: '2-digit', month: '2-digit', year: 'numeric' });
        
        // Add section header when perjumpaan changes
        if (currentPerjumpaan !== record.perjumpaan) {
          currentPerjumpaan = record.perjumpaan;
          html += `
            <tr class="bg-gradient-to-r from-yellow-800/40 to-yellow-700/40">
              <td colspan="8" class="px-4 py-3">
                <h3 class="text-lg font-bold text-yellow-500">📋 Perjumpaan Ke-${record.perjumpaan}</h3>
              </td>
            </tr>
          `;
        }
        
        html += `
          <tr class="hover:bg-gray-800/50 transition-colors">
            <td class="px-4 py-3 text-sm text-gray-300">${index + 1}</td>
            <td class="px-4 py-3 text-sm text-gray-300">${record.nama_guru}</td>
            <td class="px-4 py-3 text-sm text-gray-300">${formattedDate}</td>
            <td class="px-4 py-3 text-sm text-center">
              <span class="bg-yellow-900/50 text-yellow-500 px-3 py-1 rounded-full font-bold border border-yellow-600">${record.perjumpaan}</span>
            </td>
            <td class="px-4 py-3 text-sm text-gray-300">${record.kategori}</td>
            <td class="px-4 py-3 text-sm text-yellow-500 font-semibold">${record.aktiviti}</td>
            <td class="px-4 py-3 text-sm text-center">
              <span class="bg-yellow-600 text-black px-3 py-1 rounded-full font-bold">${record.jumlah_hadir}</span>
            </td>
            <td class="px-4 py-3 text-sm">
              <div class="flex gap-2">
                <button onclick="viewReport('${record.__backendId}')" class="bg-blue-600 hover:bg-blue-700 text-white px-3 py-1 rounded text-xs font-semibold transition-all">
                  Lihat

                  Padam
                </button>
              </div>
            </td>
          </tr>
        `;
      });
      
      tbody.innerHTML = html;
    }
