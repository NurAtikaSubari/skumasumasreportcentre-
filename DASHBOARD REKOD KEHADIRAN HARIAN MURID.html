    function renderStudents() {
      const selectedClass = document.getElementById('class-select').value;
      const container = document.getElementById('students-container');
      const noClassMessage = document.getElementById('no-class-message');
      const studentsList = document.getElementById('students-list');

      if (!selectedClass) {
        container.classList.add('hidden');
        noClassMessage.classList.remove('hidden');
        currentAttendance = {};
        return;
      }

      noClassMessage.classList.add('hidden');
      container.classList.remove('hidden');

      const students = studentLists[selectedClass] || [];
      
      if (students.length === 0) {
        studentsList.innerHTML = '<div class="text-center py-8 text-gray-500">Tiada data murid untuk kelas ini</div>';
        return;
      }

      studentsList.innerHTML = students.map((student, idx) => {
        const status = currentAttendance[idx] || 'unmarked';
        const statusClass = status === 'present' ? 'present' : status === 'absent' ? 'absent' : '';
        const statusEmoji = status === 'present' ? '‚úÖ' : status === 'absent' ? '‚ùå' : '‚≠ï';
        
        return `
          <div class="student-item ${statusClass} border rounded-lg p-3 flex items-center justify-between cursor-pointer hover:border-yellow-400/50" onclick="toggleAttendance(${idx})">
            <div class="flex-1 text-sm"><span>${student}</span></div>
            <span class="text-xl">${statusEmoji}</span>
          </div>
        `;
      }).join('');

      updateStudentStats(students.length);
    }

    function toggleAttendance(idx) {
      const status = currentAttendance[idx] || 'unmarked';
      const statuses = ['unmarked', 'present', 'absent'];
      const nextStatusIdx = (statuses.indexOf(status) + 1) % statuses.length;
      currentAttendance[idx] = statuses[nextStatusIdx];
      renderStudents();
    }

    function exportToCSV() {
      if (records.length === 0) {
        showToast('Tiada rekod untuk dieksport', 'warning');
        return;
      }

      const headers = ['Tarikh', 'Nama Guru', 'Nama Kelas', 'Jumlah Murid', 'Hadir', 'Tidak Hadir', 'Catatan'];
      const rows = records.map(r => [
        formatDate(r.date),
        r.teacher_name,
        r.class_name,
        r.total_students,
        r.present,
        r.absent,
        r.notes || ''
      ]);

      let csv = headers.join(',') + '\n';
      rows.forEach(row => {
        csv += row.map(cell => `"${cell}"`).join(',') + '\n';
      });

      const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });
      const link = document.createElement('a');
      const url = URL.createObjectURL(blob);
      const filename = `Kehadiran_${new Date().toISOString().split('T')[0]}.csv`;
      
      link.setAttribute('href', url);
      link.setAttribute('download', filename);
      link.style.visibility = 'hidden';
      document.body.appendChild(link);
      link.click();
      document.body.removeChild(link);
      
      showToast(`‚úÖ Fail CSV berjaya dimuat turun! (${filename})`);
      hideExportModal();
    }

    async function deleteRecord() {
      if (!deleteTarget || isLoading) return;
      
      const record = records.find(r => r.__backendId === deleteTarget);
      if (!record) {
        hideDeleteModal();
        return;
      }

      isLoading = true;
      document.getElementById('confirm-delete').disabled = true;
      document.getElementById('confirm-delete').textContent = 'Memadam...';

      const result = await window.dataSdk.delete(record);
      
      isLoading = false;
      document.getElementById('confirm-delete').disabled = false;
      document.getElementById('confirm-delete').textContent = 'Padam';
      
      if (result.isOk) {
        showToast('Rekod berjaya dipadam!');
      } else {
        showToast('Gagal memadam rekod', 'error');
      }
      
      hideDeleteModal();
    }
