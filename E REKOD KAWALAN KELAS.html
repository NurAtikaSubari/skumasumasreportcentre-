    async function toggleFavorite(backendId) {
      const record = records.find(r => r.__backendId === backendId);
      if (!record) return;

      const updated = { ...record, is_favorite: !record.is_favorite };
      const result = await window.dataSdk.update(updated);

      if (!result.isOk) {
        showToast('Gagal mengemas kini kegemaran', 'error');
      }
    }

      confirmBtn.disabled = false;
